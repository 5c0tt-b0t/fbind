###ONE-LINE SETUP###

# Bind-mount intsd to extsd/.fbind (includes obb)
int_extf



###MULTI-USER SETUP###

u11=/data/media/11 # specifies intsd_path for user 11.
bind_mnt $extsd/FOLDER $u11/FOLDER

u13=/data/media/13 # specifies intsd_path for user 13.
bind_mnt $extsd/FOLDER $u13/FOLDER

u# & u## are user variables -- these can be set to any value(s) you wish.



###Save /data/data/APP to intsd/.app_data or extsd/.app_data###

# To intsd/.app_data/APP (if your device hasn't or "doesn't support" SD card)
extsd_path $intsd
app_data APP

# To extsd/.app_data/APP
app_data APP



###ADVANCED SETUP###

# Auto-mount a partition & bind-mount it to $intsd/.extsd for easier access
part /dev/block/mmcblk1 /data/.extsd f2fs
bind_mnt /data/.extsd $intsd/.extsd X
obb

# Fully replace intsd with extsd (similar to the one-line setup above, but more advanced)
part /dev/block/mmcblk1 $intsd f2fs
obb

# Bind-mount intsd to extsd/.fbind
bind_mnt $extsd/.fbind $intsd -mv
obb



###MISC###

# If your SD card path is different from the standard /mnt/media_rw/* or you have more than a folder listed there, you need to specify it
extsd_path /path/to/storage

# Some users experiencing bind issues may need to to set these lines (especially true for some Samsung devices/ROMs)
intsd_path /storage/emulated/0
intobb_path /storage/emulated/obb

# Disable auto-bind service to open a LUKS volume -- handled by part() 
luks

# "pm grant" storage permissions to all apps (including future installations)
perms

# Check for/fix SD card errors before system has a chance to mount it
fsck -OPTION(s) /path/to/partition (i.e., fsck.f2fs -f /dev/block/mmcblk1)

# Disable config backup auto-restore
no_restore

# Disable config backup AND backup auto-restore
no_bkp
